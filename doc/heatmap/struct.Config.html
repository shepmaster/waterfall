Defaults

  • max_value => 1_000_000_000 (1 second in nanoseconds)
  • slice_duration => 60_000_000_000 (1 minute in nanoseconds)
  • num_slices => 60 (1 hour of heatmap)
  • start => 0 (start from time 0)
